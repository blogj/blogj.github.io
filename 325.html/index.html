version: "2"
services:
  resilio-sync:
    image: ghcr.io/linuxserver/resilio-sync
    container_name: resilio-sync
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=Asia/Shanghai
    volumes:
      - /root/resilio/config:/config
      - //mnt/sda/resilio/downloads/:/downloads
      - /mnt/sda/resilio/Sync/:/sync
    network_mode: "host"
    restart: unless-stopped
